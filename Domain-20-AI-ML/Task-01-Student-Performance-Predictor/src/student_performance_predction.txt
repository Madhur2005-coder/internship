const fields = {
  studyHours: {
    input: document.querySelector("#study-hours"),
    output: document.querySelector("#study-hours-value"),
    label: "Study time",
    formatter: (value) => `${value.toFixed(1)} h`,
    score: (value) => Math.min(value / 5, 1),
    weight: 0.22,
    advice: "Add one focused study block before increasing total hours."
  },
  attendance: {
    input: document.querySelector("#attendance"),
    output: document.querySelector("#attendance-value"),
    label: "Attendance",
    formatter: (value) => `${Math.round(value)}%`,
    score: (value) => value / 100,
    weight: 0.2,
    advice: "Prioritize attendance; missed classes quickly lower the estimate."
  },
  sleepHours: {
    input: document.querySelector("#sleep-hours"),
    output: document.querySelector("#sleep-hours-value"),
    label: "Sleep",
    formatter: (value) => `${value.toFixed(1)} h`,
    score: (value) => Math.max(0, 1 - Math.abs(value - 7.5) / 4.5),
    weight: 0.14,
    advice: "Aim for a consistent 7 to 8 hours of sleep."
  },
  participation: {
    input: document.querySelector("#participation"),
    output: document.querySelector("#participation-value"),
    label: "Participation",
    formatter: (value) => `${Math.round(value)} / 10`,
    score: (value) => value / 10,
    weight: 0.13,
    advice: "Ask or answer more questions during class to reinforce recall."
  },
  assignments: {
    input: document.querySelector("#assignments"),
    output: document.querySelector("#assignments-value"),
    label: "Assignments",
    formatter: (value) => `${Math.round(value)}%`,
    score: (value) => value / 100,
    weight: 0.14,
    advice: "Complete pending assignments before adding extra revision."
  },
  previousScore: {
    input: document.querySelector("#previous-score"),
    output: document.querySelector("#previous-score-value"),
    label: "Previous score",
    formatter: (value) => `${Math.round(value)}%`,
    score: (value) => value / 100,
    weight: 0.17,
    advice: "Review mistakes from the previous exam to lift the baseline."
  }
};

const predictedScore = document.querySelector("#predicted-score");
const performanceBand = document.querySelector("#performance-band");
const scoreFill = document.querySelector("#score-fill");
const factorList = document.querySelector("#factor-list");
const suggestion = document.querySelector("#suggestion");

function clamp(value, min, max) {
  return Math.min(Math.max(value, min), max);
}

function getBand(score) {
  if (score >= 90) return "Excellent readiness";
  if (score >= 80) return "Strong readiness";
  if (score >= 70) return "Good readiness";
  if (score >= 60) return "Developing readiness";
  return "At-risk readiness";
}

function calculatePrediction() {
  const factors = Object.entries(fields).map(([key, field]) => {
    const rawValue = Number(field.input.value);
    const normalized = clamp(field.score(rawValue), 0, 1);

    field.output.value = field.formatter(rawValue);

    return {
      key,
      label: field.label,
      value: rawValue,
      normalized,
      weightedScore: normalized * field.weight,
      gap: (1 - normalized) * field.weight,
      advice: field.advice
    };
  });

  const weightedTotal = factors.reduce((sum, factor) => sum + factor.weightedScore, 0);
  const predicted = Math.round(clamp(45 + weightedTotal * 55, 0, 100));

  predictedScore.textContent = predicted;
  performanceBand.textContent = getBand(predicted);
  scoreFill.style.width = `${predicted}%`;

  const rankedFactors = [...factors].sort((a, b) => b.weightedScore - a.weightedScore).slice(0, 3);
  factorList.innerHTML = rankedFactors.map((factor) => {
    const percent = Math.round(factor.normalized * 100);
    return `
      <li>
        <div class="factor-row">
          <span>${factor.label}</span>
          <span>${percent}%</span>
        </div>
        <div class="factor-bar"><span style="width: ${percent}%"></span></div>
      </li>
    `;
  }).join("");

  const biggestGap = [...factors].sort((a, b) => b.gap - a.gap)[0];
  suggestion.textContent = biggestGap.advice;
}

Object.values(fields).forEach((field) => {
  field.input.addEventListener("input", calculatePrediction);
});

calculatePrediction();
